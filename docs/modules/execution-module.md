        # 猫聛職氓聬聢莽禄聯忙聻聹
        return self._aggregate_results(results)
    
    async def _execute_sequential(
        self,
        plan: ExecutionPlan,
        executor: ExecutorAgent
    ) -> ExecutionResult:
        """盲赂虏猫隆聦忙聣搂猫隆聦"""
        
        results = []
        
        for task in plan.tasks:
            # 莽颅聣氓戮聟盲戮聺猫碌聳氓庐聦忙聢聬
            await self._wait_for_dependencies(task, plan.dependencies)
            
            # 忙聣搂猫隆聦盲禄禄氓聤隆
            result = await self._execute_task_with_retry(task, executor)
            results.append(result)
            
            # 氓娄聜忙聻聹盲禄禄氓聤隆氓陇卤猫麓楼盲赂聰盲赂聧氓聟聛猫庐赂莽禄搂莽禄颅茂录聦氓聛聹忙颅垄忙聣搂猫隆?            if not result.success and not self.config.get("continue_on_error"):
                break
        
        return self._aggregate_results(results)
    
    async def _execute_hybrid(
        self,
        plan: ExecutionPlan,
        executor: ExecutorAgent
    ) -> ExecutionResult:
        """忙路路氓聬聢忙聣搂猫隆聦 - 忙聶潞猫聝陆茅聙聣忙聥漏忙聣搂猫隆聦忙聳鹿氓录聫"""
        
        # 氓聢聠忙聻聬盲禄禄氓聤隆盲戮聺猫碌聳氓聸?        dependency_graph = self._build_dependency_graph(plan)
        
        # 猫炉聠氓聢芦氓聫炉盲禄楼氓鹿露猫隆聦忙聣搂猫隆聦莽職聞莽禄聞
        parallel_groups = self._identify_parallel_groups(dependency_graph)
        
        # 忙聦聣莽禄聞忙聣搂猫隆聦
        results = []
        for group in parallel_groups:
            if len(group) == 1:
                # 氓聧聲盲赂陋盲禄禄氓聤隆茂录聦盲赂虏猫隆聦忙聣搂猫隆?                result = await self._execute_task_with_retry(group[0], executor)
                results.append(result)
            else:
                # 氓陇職盲赂陋盲禄禄氓聤隆茂录聦氓鹿露猫隆聦忙聣搂猫隆?                group_results = await asyncio.gather(*[
                    self._execute_task_with_retry(task, executor)
                    for task in group
                ])
                results.extend(group_results)
        
        return self._aggregate_results(results)

async def _execute_task_with_retry(
    self,
    task: Task,
    executor: ExecutorAgent
) -> TaskResult:
    """忙聣搂猫隆聦盲禄禄氓聤隆茂录聦忙聰炉忙聦聛茅聡聧猫炉?""
    
    for attempt in range(task.max_retries + 1):
        try:
            # 忙聣搂猫隆聦盲禄禄氓聤隆
            result = await executor.execute(task)
            
            return TaskResult(
                task_id=task.id,
                success=True,
                result=result,
                attempts=attempt + 1
            )
            
        except Exception as e:
            task.retry_count += 1
            
            if task.retry_count > task.max_retries:
                # 茅聡聧猫炉聲忙卢隆忙聲掳莽聰篓氓掳陆茂录聦氓掳聺猫炉聲氓聢聡忙聧垄忙聶潞猫聝陆盲陆聯
                if self.fault_tolerance:
                    new_agent = await self._switch_agent(task)
                    if new_agent:
                        task.assigned_agent = new_agent
                        task.retry_count = 0
                        continue
                
                return TaskResult(
                    task_id=task.id,
                    success=False,
                    error=str(e),
                    attempts=attempt + 1
                )
            
            # 莽颅聣氓戮聟氓聬聨茅聡聧猫炉?            await asyncio.sleep(self.config.get("retry_delay", 1.0) * (2 ** attempt))
    
    return TaskResult(
        task_id=task.id,
        success=False,
        error="Max retries exceeded",
        attempts=task.retry_count
    )
